OLD: Copy paste below in above file:
#!/bin/bash
sudo kill -9 $(lsof -ti :8082)


New:
#!/bin/bash
PORT=8082

# Find PID of process using port 8082
PID=$(sudo lsof -ti :$PORT)

if [ -z "$PID" ]; then
  echo "❌ No Spring Boot app running on port $PORT"
  exit 0
fi

echo "🛑 Sending SIGTERM to process $PID for graceful shutdown..."
sudo kill -15 $PID

# Wait up to 20 seconds for graceful shutdown
for i in {1..40}; do
  if ps -p $PID > /dev/null; then
    echo "⏳ Waiting for app to stop... ($i sec)"
    sleep 1
  else
    echo "✅ App stopped gracefully."
    exit 0
  fi
done

echo "⚠️ App did not stop in time, forcing shutdown..."
sudo kill -9 $PID
echo "💀 Forced stop completed."

Step 4 — Setup Logging
sudo journalctl --unit=qht-voicebridge

See realtime logs using -f option.
Use -n <# of lines> to view specified number of lines of log
sudo journalctl -f -n 1000 -u qht-voicebridge

Tail the live log using -f option
sudo journalctl -f -u qht-voicebridge

See Logs After specific Time
sudo journalctl -f -u qht-voicebridge --since "2025-09-26 15:40:00"

Stop the service
sudo systemctl stop qht-voicebridge
